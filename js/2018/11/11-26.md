    // 获取地址的参数并转成对象

    function  getURLParameters(url)  {
      return (url.match(/([^?=&]+)(=[^&]*)/g) || []).reduce((a, v) => 
      ((a[v.slice(0, v.indexOf('='))] = v.slice(v.indexOf('=') + 1)), a), {})
    }
  console.log(getURLParameters('www.shamopoo.top?name=shamoppoo&sex=1&age=3'))
